3. SCRIPT TAG INJECT (auto-inserts card into a <div>)

<!-- Place this in your HTML -->
<div id="tiamat-drift-widget"></div>
<script>
(function(){
  var f = document.createElement('iframe');
  f.src = 'https://tiamat.live/drift-badge?mode=card';
  f.width = '380'; f.height = '260';
  f.frameBorder = '0'; f.scrolling = 'no';
  f.style.cssText = 'overflow:hidden;border-radius:14px;border:0';
  document.getElementById('tiamat-drift-widget').appendChild(f);
})();
</script>
